Custom validation (Advanced use case)

Showing custom validation for form components in the GC Design System.


<gcds-input
  input-id="form-name"
  label="Name of component library"
  hint="Please enter '@gcds-core/components' to continue."
  name="library-name"
  required
  size="20"
></gcds-input>

<script>
  const input = document.querySelector('gcds-input[name="library-name"]');

  const libraryName = () => {
    return {
      validate: value => {
      return {
        valid: value === '@gcds-core/components',
        reason: {
          en: `The entered value must be '@gcds-core/components'.`,
          fr: `La valeur saisie doit être « @gcds-core/components ».`,
        },
      };
      },
    };
  };

  input.validator = [libraryName()];
</script>



<gcds-checkboxes
  legend="Checkboxes example"
  name="checkexample"
  hint="Please select only two checkboxes to continue."
  required
  options='[
    {"id": "form-check-1", "label": "Checkbox 1", "value": "check1"},
    {"id": "form-check-2", "label": "Checkbox 2", "value": "check2"},
    {"id": "form-check-3", "label": "Checkbox 3", "value": "check3"},
    {"id": "form-check-4", "label": "Checkbox 4", "value": "check4"}
  ]'>
</gcds-checkboxes>

<script>
  const checkboxes = document.querySelector('gcds-checkboxes[name="checkexample"]');

  const checkboxesValidator = () => {
    return {
      validate: value => {
        return {
          valid: value.length === 2,
          reason: {
            en: `Please select two checkboxes to continue.`,
            fr: `Veuillez sélectionner deux cases à cocher pour continuer.`,
          },
        };
      },
    };
  };

  checkboxes.validator = [checkboxesValidator()];
</script>

<script>
  const checkboxesTwo = document.querySelector('gcds-checkboxes[name="checkexampletwo"]');

    };

  checkboxesTwo.validator = [selectAllValidator(4)];
